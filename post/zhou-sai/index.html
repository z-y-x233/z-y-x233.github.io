A 递归式

题目描述

fn={an=0bn=1(1+fn−1)fn−2n>1f_n = \begin{cases} a & n = 0 \\b & n = 1 \\ \frac{(1 + f_{n-1})} {f_{n-2}} & n > 1\end{cases} fn​=⎩⎪⎨⎪⎧​abfn−2​(1+fn−1​)​​n=0n=1n>1​

给定a,b,na,b,na,b,n， 求fnf_nfn​

分析

题目看着挺唬人的，稍微推导一下就会发现

f2=1+ba,f3=1+a+ba+b,f4=1+ba,f5=a=f0f_2 = \frac{1 + b}{a}, f_3 = \frac{1 + a + b}{a +b},f_4 = \frac{1 + b}{a},f_5 = a = f_0 f2​=a1+b​,f3​=a+b1+a+b​,f4​=a1+b​,f5​=a=f0​

所以其实就是个循环，没啥好解释的

代码

#include<bits/stdc++.h>

using namespace std;

int main() {
	int t;
	scanf("%d",&t);
	while (t--) {
		double a, b, res;
		int n;
		scanf("%lf%lf", &a, &b);
		scanf("%d",&n);
		if (n % 5 == 0) res = a;
		else if (n % 5 == 1) res = b;
		else if (n % 5 == 2) res = (1.0 + b) / a;
		else if (n % 5 == 3) res = (1.0 + a + b) / (a + b);
		else if (n % 5 == 4) res = (1.0 + a) / b;
		printf("%.3f\n", res);
 	}

	return 0;
}

B 旅行

题目描述

lcllcllcl打算去nnn个城市旅行，他的预算是WWW

城市编号从1−n1−n1−n，城市与城市之间两两可达

城市uuu到城市vvv的花费为wu,vw_{u,v}wu,v​

lcllcllcl从城市111出发，他的目标是除了出发的城市111外，每个城市都到达并且只到达一次。在最后回到出发的城市，并且恰好用完预算，求一共有多少种不同的旅行方案

题目保证至少有一个解

分析

没啥好分析的，由于题目城市数量最大只有8，所以我们直接暴搜就完事了

解法一

因为题目要求的是从111出发并在最终回到111，所以起点和终点都为111，又因为其余每个点都只经过了一次。

那么对于任意nnn个点的路径来说起点与中点之间的所有路径就是2−n2 - n2−n的全排列

所以我们只需要枚举出2−n2 - n2−n的所有排列再计算出这条路径的花费，当当前路径花费等于 WWW 时，让最终答案加一，即可解决

那么这个时候问题来了，如何求一个序列的全排列

答案是 手写一个

next_permutation()

首先我们要知道C++有一个特别神奇的库<\lt< algorithm>\gt>，这个库里面有很多神奇的函数

比如说我们的 next_permutation()， 他的作用是按字典序求一个序列的下一个字典序更大的排列，所以初始序列必须是字典序最小的，否则无法生成所有排列。

这个时候有人说了，没学c++不知道这个函数怎么办， 你可以手写一个啊

int next_permutation(int * start, int *end) {
	int * p2 = end - 1;
	while (p2 >= start && (*p2 < *(p2 - 1))) p2--;
	if (p2 == start) return 0;
	int * p1 = p2 - 1;
	int * p3 = end - 1;
	while (*p3 < *p1) p3--;
	swap(*p3, *p1);
	int * l = p2, *r = end - 1;
	while (l < r) 
		swap(*l++, *r--);
	return 1;
}

代码

#include<stdio.h>

void swap(int * a,int * b){
    int t = *a;
    *a = *b;
    *b = t;
}

int next_permutation(int * start, int *end) {
	int * p2 = end - 1;
	while (p2 >= start && (*p2 < *(p2 - 1))) p2--;
	if (p2 == start) return 0;
	int * p1 = p2 - 1;
	int * p3 = end - 1;
	while (*p3 < *p1)p3--;
	swap(p3, p1);
	int * l = p2, *r = end - 1;
	while (l < r) 
		swap(l++, r--);
	return 1;
}

int main() {
	int n, s;
	scanf("%d%d",&n,&s);
	int w[10][10];
	for (int i = 1; i <= n; ++i) {
		for (int j = 1; j <= n; ++j) {
			scanf("%d", &w[i][j]);
		}
	}
	int res = 0;
	int a[10] = { 1,2,3,4,5,6,7,8 };
	do {
		int sum = 0;
		int la = 1;
		for (int i = 1; i < n; ++ i) {
			sum += w[la][a[i]];
			la = a[i];
		}
		sum += w[la][1];
		if (sum == s) res++;
	} while (next_permutation(a + 1, a + n));
	printf("%d",res);
	return 0;
}

解法二

爆搜，用dfsdfsdfs枚举一下每次经过了哪个点，走完后判断一下是否满足条件就行，dfsdfsdfs其实就是一个枚举全排列的过程

代码

#include<stdio.h>

int n, W;
int w[10][10];
int res = 0;

//参数分别为当前已经走过的点，当前花费，当前在哪个城市
void dfs(int sta, int sum, int cur) {
	//代表已经走完所有的点
	if (sta == ((1 << n) - 1)) {
		//最后需要回到第一个点
		if (sum + w[cur][1]== W) res++;
		return;
	}
	for (int i = 2; i <= n; ++i) {
		//如果当前枚举到的点没有走过那就走一下
		if ((sta >> (i - 1) & 1) == 0) {
			dfs(1 << (i - 1) | sta, sum + w[cur][i], i);
		}
	}
}

 int main() {
	 scanf("%d%d", &n, &W);
	 for (int i = 1; i <= n; ++i)
		 for (int j = 1; j <= n; ++j)
			 scanf("%d", &w[i][j]);
	 dfs(1, 0, 1);
	 printf("%d", res);
	return 0;
}

C 替换

题目描述

给定一个长度为nnn的整数序列，对这个序列进行qqq次操作

每次操作给出两个数a,ba, ba,b，代表把序列中的aaa全部替换为bbb

因为我们需要求的答案为数列和，而对于每一次的操作 ,会把整个数组和和sumsumsum减去aaa的数量乘aaa再加上aaa的数量乘bbb

所以我们只需要求得他们的差值并且修改他们就行

而因为数据范围为1e51e51e5级别的数，所以直接暴力修改替换肯定会超时

这时我们可以注意到a,ba,ba,b的范围也为1e51e51e5，那么我们可以利用桶排的思想

建立一个$$ n$$大于等于1e51e51e5数组sns_nsn​,把a,ba,ba,b直接做为数组下标

用sxs_xsx​表示xxx这个数出现了多少次，在读入中每次读入一个xxx，就把sxs_xsx​加一

那么这样存储数据后，对于每一次操作qqq我们可以令sum=sum−sa×a+sa×bsum =sum - s_a \times a + s_a \times bsum=sum−sa​×a+sa​×b

然后修改a,ba,ba,b的数量,sb+=sa,sa=0s_b += s_a, s_a = 0sb​+=sa​,sa​=0

对于a=ba = ba=b的情况我们需要特判一下直接输出

这样对于每一次操作qqq，只需要花费O(1)O(1)O(1)的时间，对于每次操作后输出sumsumsum即可

代码

#include<stdio.h>
#define N 100005
int e[N];
long long sum = 0;

int main(void) {
	int  n, q, x, a, b;
	scanf("%d%d",&n,&q);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &x);
		e[x]++;
		sum += x;
	}
	while (q--) {
		scanf("%d%d", &a, &b);
		if (a == b) {
			printf("%lld\n", sum);
			continue;
		}
		sum -= e[a] * a;
		e[b] += e[a];
		sum += e[a] * b;
		e[a] = 0;
		printf("%lld\n", sum);
	}
}

E 区间和

题目描述

给定一个长度为n的序列a,试求出a的一段非空子区间,使得区间内所有元素的和为n的整数倍

数据范围

1≤n≤50001 \le n \le 50001≤n≤5000

1≤ai≤50001 \le a_i \le 50001≤ai​≤5000

分析

因为简单版本的数据范围较小，我们可以直接枚举所有的[i,j][i,j][i,j]区间，然后每次判断是否满足条件即可，而因为我们是从小到大枚举，所以满足条件的第一个区间就是字典序最小的区间。并且在加强版中会证明一定存在一个区间是满足条件的

代码

#include<stdio.h>

int main(){
	int n,a[5005];
    scanf("%d",&n);
    //这里因为区间下边是从1开始，所以存储的时候下标从1开始也更加方便
    for (int i = 1; i <= n; ++i){
        scanf("%d",&a[i]);
    }
    for (int i = 1; i <= n; ++i){
        int sum = 0;
        for (int j = i; j <= n; ++j){
            //在这里我们可以直接在枚举右端点范围的时候把sum累加起来计算
            //如果是每次先枚举左右端点然后再遍历计算的话，时间复杂度就是O(n^3),一定会超时
            sum = (sum + a[j]) % n;
            if(sum == 0){
                printf("%d %d",i,j);
                //如果找到了就直接return掉，不需要开变量记录了，省代码
                return 0;
            }
        }
    }
	return 0;
}

F 区间和（加强版）

加强版唯一的不同就是nnn的数据范围最大变成了1e51e51e5，在1e51e51e5的数据范围下，使用简单版本的O(n2)O(n^2)O(n2)的算法也一样会超时。

那么这个时候我们就需要考虑使用其他的算法，我们注意到，题目的要求是，求得一段区间，使得他们的和是nnn的整数倍。

我们可以从这个整数倍来思考。一个数是nnn的整数倍，等价于这个数模nnn的余数为0。换句话来说，我们需要找到aaa的一个非空子区间，并且这个区间里所有元素的和sum≡0(mod n)sum \equiv 0(mod\,n)sum≡0(modn)。

当我们需要快速求一段区间和的时候，我们通常都会优先考虑一下前缀和。

前缀和上次周赛已经涉及过，这里再重复一下

定义sns_nsn​为ana_nan​的前缀和数组

对与任意的si(1≤i≤n)s_i(1 \le i \le n)si​(1≤i≤n)满足条件si=a1+a2+a3+⋯+ai=∑j=1iajs_i = a_1 + a_2 + a_3 + \cdots +a_i = \sum_{j = 1}^{i}a_jsi​=a1​+a2​+a3​+⋯+ai​=∑j=1i​aj​

那么通过这样的处理后，如果我们想求一段区间aaa中[l,r][l,r][l,r]的和，只需要之间令s[r]−s[l−1]s[r] - s[l - 1]s[r]−s[l−1]即是结果

那么正常条件下，如果aaa的一段区间[l,r][l,r][l,r]的和为零，那么等价于在它的前缀和数组sns_nsn​中s[r]=s[l−1]s[r] = s[l - 1]s[r]=s[l−1]

而在模nnn的条件下，我们只需要让s[r] % n=s[l−1] % ns[r]\,\%\, n = s[l - 1]\,\%\,ns[r]%n=s[l−1]%n，只要满足这个条件那么[l,r][l,r][l,r]这段区间的和在模nnn的条件下就是0

通过以上的分析，我们可以发现，题目的要求等价于找到ana_nan​的两个不同的前缀和在模nnn后是相等的

然后我们神奇的发现，一个长度为nnn的序列一共有nnn个前缀和，而一个数在模nnn之后最多只有n−1n - 1n−1个不为000的数，一定有解

所以我们只需要求出aaa模nnn后的前缀和，然后直接枚举寻找相同或者为000的前缀和即可，需要注意就是找最小字典序需要特殊处理一下(要是有spj就没有这个条件了)

时间复杂度O(n)O(n)O(n)

代码

#include<iostream>
#include<string>
using namespace std;

const int N = 1e5 + 100;

//三个数组意义分别为原数组，标记数组
//标记数组的作用是标记某个前缀和是否出现，如果出现记录他第一次出现的下标
int a[N], s[N];
int minl, minr;
int main() {
	
	int n, l, r;
	scanf("%d", &n);
    //先让答案区间最大化
    minl = minr = n;
    //我们初始化标记数组为-1来表示一个数没有出现过，我们让0第一次出现的下标为0
    //这样处理后，如果一段前缀和a[r]本身就为0，那么我们也能同时求出区间是[1,r]
	for (int i = 1; i < n; ++i) s[i] = -1;
	for (int i = 1; i <= n; ++i) {
		scanf("%d", &a[i]);
        //这里我们不开新的数组，直接令a为它的前缀和数组
		a[i] = (a[i] + a[i - 1]) % n;
	}
	for (int i = 1; i <= n; ++i) {
        //前缀和a[i]首次出现的下标，如果没有出现过，则为0
        int l = s[a[i]];
        //如果一个前缀和没有标记过，记录它的下标
		if(l == -1) s[a[i]] = i;
        else if(l + 1 < minl){
            //否则如果当前前缀和的左端点比当前答案区间小就更新答案区间
            //因为是从小到大枚举区间，所以如果相等右端点不可能比答案的右端点小，一定不会被更新
            minl = l + 1;
            //这里需要注意，因为当前是前缀和a[l]和a[i]相等
            //所以我们的到的为0的区间其实是[l + 1, i]
            minr = i;
        }
        
	}
    //最后输出答案即可
    printf("%d %d",minl,minr);
	return 0;
}
